class Solution {
    public int searchInsert(int[] nums, int target) 
    {
        int low=nums[0];
        int high=nums[nums.length-1];
        int mid=low+(high-low)/2;;
        while(low<=high)
        {
            if(nums[mid]==target)
                return mid;
            else if(nums[mid]<target)
                low=mid+1;
            else
                high=mid-1;
        }
        return low;
    }
}
